## Задание
- Проверьте, что код дуальный коду Хэмминга действительно является симплексом.

---

## Решение
```python
import itertools


def check_simplex_code(r):
    """
    Создает дуальный код Хэмминга порядка r и проверяет,
    является ли он симплексным (все ненулевые слова имеют одинаковый вес).
    """
    # Вычисляем длину кодового слова: n = 2^r - 1
    n = (1 << r) - 1

    # Ожидаемый вес Хэмминга для ненулевых слов: 2^(r-1)
    expected_weight = 1 << (r - 1)

    # Дуальную матрицу Хэмминга(числа от 1 до n в двоичном представлении).
    generator_matrix_cols = []
    for i in range(1, n + 1):
        binary_str = format(i, f'0{r}b')
        col = [int(bit) for bit in binary_str]
        generator_matrix_cols.append(col)

    # Все возможные исходные сообщения длины r
    messages = list(itertools.product([0, 1], repeat=r))

    print(f"--- Проверка для порядка r = {r} ---")
    print(f"Длина кодового слова (n): {n}")
    print(f"Ожидаемый вес ненулевых слов: {expected_weight}\n")

    is_simplex = True

    # Проверяем вес - умножем каждое сообщение на матрицу
    for msg in messages:
        codeword = []

        for col in generator_matrix_cols:
            bit = sum(m * c for m, c in zip(msg, col)) % 2
            codeword.append(bit)

        weight = sum(codeword)

        if weight == 0:
            print(f"Сообщение {msg} -> Кодовое слово {codeword} | Вес: {weight} (Нулевое слово)")
        else:
            print(f"Сообщение {msg} -> Кодовое слово {codeword} | Вес: {weight}")

            # Проверка на симплекс
            if weight != expected_weight:
                is_simplex = False

    print("\n" + "=" * 40)
    if is_simplex:
        print("Все ненулевые кодовые слова имеют одинаковый вес - Дуальный код Хэмминга действительно является Симплексным")
    else:
        print("ОШИБКА: Веса различаются. Код не является симплексным.")
    print("=" * 40)

for n in [3,4,5]:
    check_simplex_code(n)
```
